index.js

    hljs.initHighlighting();

    Reveal.initialize({
        width: "100%",
        height: "100%",

        controls: false,
        progress: false,
        slideNumber: true,
        history: true,
        theme: "white"
    });

});

            </div>

              <pre class="html"><code>

Hover Rules - the CSS


.dropdown {
    list-style-type: none;
    margin: 0;
    padding: 0;
}

.dropdown li {
    display: inline;
}

.dropdown li div {
    list-style-type: none;
    margin: 0;
    padding: 0;
    display: none;
    position: absolute;
}

.dropdown li:hover div {
    display: block;
    text-align: center;
}
