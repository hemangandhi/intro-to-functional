Example Problem

Problem: Given a string $s$ and nonnegative integer $k$ find the longest (or any longest, if multiple) substring of $s$ such that there are at most $k$ distinct characters.

For example, $f(s=\text{aabcdef}, k=3) = \text{aabc}$

Example Problem


          function f(str, k) {
            if ( str.length == 0 ) {
              return "";
          }

          let lengths = [];

          for (let i=0;i<str.length;i++) {
            let dict = new Map();
            let distinct = 0;

            let j;
            for (j=0;j+i<str.length;j++) {
              if (!dict.get(str[i+j])) {
                if ( distinct == k ) {
                  break;
                }

                distinct++;
                dict.set(str[i+j], true);
              }
            }

            lengths.push(j);
          }

          let maxPos = findMaxIndex(lengths);
          return str.substr(maxPos, lengths[maxPos]);
        }

        function findMaxIndex(arr) {
          let maxIndex = 0;

          for (let i=1;i<arr[i];i++) {
            if (arr[i] > arr[maxIndex]) {
              maxIndex = i;
            }
          }

          return maxIndex;
        }

Factorial

Problem: Write a function $f$ that, given a nonnegative integer returns an integer such that

$$f(0)=0 \\ f(1)=1 \\ f(n)=f(n-1)+f(n-1)$$

          function f(n) {
            let prev = 0;
            let current = 1;

            if ( n == 0 ) {
              return prev;
            }

            for (var i=0;i<n-1;n++) {
              let temp = prev;
              prev = current;
              current = prev+temp;
            }

            return current;
          }
